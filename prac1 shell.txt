{

}

insert()
{
db="emp";
echo "******insert employee record**********";
echo "enter the emp_id";
read id
echo "enter the name of employee";
read name
echo "enter the location or city";
read city
echo "enter the salary of employee";
read sal
tuple=" |$id || $name || $city || $sal |";
echo $tuple >> $db;
}

view()
{
db="emp";
echo "******display database**********";
cat $db;
}

search()
{
db="emp";
echo "******search employee record from database**********";
echo "enter the name of employee to search hisd/her record";
read name
if grep "$name" "$db"
then 
echo "record is found";
else
echo "record is not found";
fi
}
delet()
{
db="emp";
echo "******delete record form database**********";
echo "enter the name of employee we want to delete";
read name
if grep "$name" "$db"
# use grep -v for insert matching and mv the content in database
# use sed command to select record and delete it
then
#sed-i /$name/d $db
grep -v "$name" "$db">temp
mv temp $db
echo "record been deleted"
else
echo "invalid record for delete"
fi
}
modify()
{
db="emp";
echo "******modify employee record in database**********";
echo "enter the name of employee whose record you want to modify";
read name
if grep -i $name $db
then
grep -v $name $db > temp
mv temp $db
echo "enter modified name, city, id and sal again"
read name city id sal
record=" |$id || $name || $city || $sal |";
echo $record >> $db;
echo "record modified successfully and modified record is ";
echo "$record";
else
echo "record you want to modify is not present please enter valis one";
fi
}
